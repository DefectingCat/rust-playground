```rust
use std::{sync::mpsc, thread, time::Duration};

fn main() {
    let (tx, rx) = mpsc::channel();

    let tx1 = tx.clone();
    thread::spawn(move || {
        let msgs = vec![String::from("hi"), String::from("hi1"), String::from("hi2")];

        let _: Vec<_> = msgs
            .into_iter()
            .map(|msg| {
                tx1.send(msg).unwrap();
                thread::sleep(Duration::from_secs(1))
            })
            .collect();
    });

    thread::spawn(move || {
        let vals = vec![
            String::from("more"),
            String::from("messages"),
            String::from("for"),
            String::from("you"),
        ];

        for val in vals {
            tx.send(val).unwrap();
            thread::sleep(Duration::from_secs(1));
        }
    });

    rx.iter().map(|r| println!("Got: {}", r)).collect()
}
```
